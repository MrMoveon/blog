# Electron 5.0+

Electron是由Github开发，用HTML，CSS和JavaScript来构建跨平台桌面应用程序的一个开源库。 Electron通过将Chromium和Node.js合并到同一个运行时环境中，并将其打包为Mac，Windows和Linux系统下的应用来实现这一目的。

以下示例github项目地址

# 配置开发环境

Windows 开发环境配置

安装最新版本的 == Node.js == 。推荐安装最新的 长期支持版本 或者 当前发行版本 ，访问 nodejs官网，选择Windows Installer。 下载完成后， 执行安装程序，根据引导完成安装即可。

main.js

const { app, BrowserWindow } = require("electron");
// 创建一个主窗口
let mainWindow = null;

function createWindow() {
  mainWindow = new BrowserWindow({
    width: 800,
    height: 600,
    webPreferences: {
      nodeIntegration: true
    }
  });
  // 载入模板文件
  mainWindow.loadFile("./index.html");
  // 打开开发调试工具
  mainWindow.webContents.openDevTools();

  mainWindow.on("closed", () => {
    mainWindow = null;
  });
}

// app 准备好，创建window
app.on("ready",createWindow);

app.on("window-all-closed",()=>{
    // 在macOS上，应用程序及其菜单栏是常见的
    // 在用户显式退出Cmd + Q之前保持活动状态
    if (process.platform !== 'darwin') app.quit()
})
app.on("activate",()=>{
    // 在mac上，通常会在应用程序中重新创建一个窗口
    // 单击dock图标，没有其他窗口就打开
    if(!mainWindow) createWindow()
})

在最新的electron 5.x版本中默认没法在渲染进程引入模块，渲染进程会提示 require is not defined,解决方案：需要在BrowserWindow属性里配置webPreferences

 mainWindow = new BrowserWindow({
    width: 800,
    height: 600,
    // 必须加上以下代码
    webPreferences: {
      nodeIntegration: true
    }
  });

# 拖放并打开文件

render.js

const fs = require("fs");

var area = document.querySelector("#drag-area");

// 阻止拖动区域开始拖动，拖动中的默认事件
area.addEventListener("dragover",function(e){
    e.preventDefault()
})
area.addEventListener("dragstart",function(e){
    e.preventDefault()
})
// 监听拖动放置事件，如果是文本，写入到拖动区域
area.addEventListener("drop",function(e){
    
    // 判断文件类型
    console.log(e.dataTransfer.items[0].type);
    var type = e.dataTransfer.items[0].type;
    
    // 文件路径
    var path = e.dataTransfer.files[0].path

    // 读取文件，并插入
    fs.readFile(path,'utf-8',(err,data)=>{
        if(err) return console.log(err);
        checkType(type,path,data)
    })
})
// 检测文件类型，并插入到area
function checkType(type,path,data){
    if(type.indexOf("image")>-1){
       var img = document.createElement('img');
       img.src =path;
       area.appendChild(img)
    }
    if(type.indexOf("text")>-1){
        area.innerHTML = data
    }
}

# remote模块

remote 模块提供了一种在渲染进程（网页）和主进程之间进行进程间通讯（IPC）的简便途径。

Electron 中, 与 与 GUI 块 相关的模块 （如 如 dialog, menu 等) 只存在于主进程中而不在渲染进程中 。

为了能从渲染进程中使用它们，需要用ipc模块来给主进程发送进程间消息。使用 remote 模块，可以调用主进程对象的方法，而无需显式地发送进程间消息。

# remote 打开新窗口

在渲染进程中，可以直接通过remote模块打开窗口

const { BrowserWindow } = require("electron").remote;
var btn = document.querySelector("#btn");
var win = null;
btn.addEventListener("click", function() {
  win = new BrowserWindow({
    width: 400,
    height: 400,
    frame:false,// 无边框窗口
    fullscreen:false,// 打开全屏显示窗口
    show: false, // 打开窗口不显示
    // backgroundColor: '#2e2c29', //窗口背景色
    webPreferences:{
      nodeIntegration: true
    }
  });
  // 载入模板
  win.loadFile("./index.html");
  // 渲染进程完成绘制时,显示窗口，窗口将没有视觉闪烁
  win.once('ready-to-show', () => {
    win.show()
  })
  win.on("closed", function() {
    win = null;
  });
});

# 优雅地显示窗口

当页面在窗口中直接加载时，用户会看到未完成的页面，这不是一个好的原生应用的体验。为了让画面准备好了再显示，这有两种不同的解决方案。

# 使用ready-to-show事件

在加载页面时，渲染进程第一次完成绘制时，会发出 ready-to-show 事件 。 在此事件后显示窗口将没有视觉闪烁：

// 创建窗口 默认不显示
let win = new BrowserWindow({ show: false })
// 窗口加载完毕，显示
win.once('ready-to-show', () => {
  win.show()
})
# 设置 backgroundColor

对于一个复杂的应用，ready-to-show 可能发出的太晚，会让应用感觉缓慢。 在这种情况下，建议立刻显示窗口，并使用接近应用程序背景的 backgroundColor

const { BrowserWindow } = require('electron')
let win = new BrowserWindow({ backgroundColor: '#2e2c29' })
win.loadURL('https://github.com')

请注意，即使是使用 ready-to-show 事件的应用程序，仍建议使用设置 backgroundColor 使应用程序感觉更原生

# 进程通信

# 主进程和渲染进程通信

有时候我们想在渲染进程中通过一个事件去执行主进程里面的方法。或者在渲染进程中通知主进程处理事件，主进程处理完成后广播一个事件让渲染进程去处理一些事情。这个时候就用到了主进程和渲染进程之间的相互通信。

Electron 主进程，和渲染进程的通信主要用到两个模块：ipcMain 和 和 ipcRenderer

ipcMain ： 当在主进程中使用时，它处理从渲染器进程（网页）发送出来的异步和同步信息,当然也有可能从主进程向渲染进程发送消息。

ipcRenderer： 使用它提供的一些方法从渲染进程 (web 页面) 发送同步或异步的消息到主 进程。 也可以接收主进程回复的消息。

场景 1 ： 渲染进程给主进程发送异步消息

场景 2 ： 渲染进程给主进程发送异步消息，主进程接收到异步消息以后通知渲染进程：

主进程

const {ipcMain } = require("electron");

ipcMain.on("message",(event,data)=>{
    console.log(data);
})
ipcMain.on("msg",(event,data)=>{
    console.log("主进程接收到渲染进程的消息：");
    console.log(data);
    console.log("通知渲染进程可以开始了");
    event.sender.send("start",true)
})

渲染进程

const {ipcRenderer} = require("electron")

var btn1 = document.querySelector("#btn1")
var btn2 = document.querySelector("#btn2")


btn1.addEventListener("click",function(){
  ipcRenderer.send("message","渲染进程给主进程发送了一个消息")
})

btn2.addEventListener("click",function(){
  ipcRenderer.send("msg",{name:"李四"})
})

ipcRenderer.on("start",(event,data)=>{
  console.log("主进程返回渲染进程的消息:"+ data);
  
})

# 渲染进程和渲染进程通信

方式一：通过localstorage给另一个渲染进程传值

localStorage.setItem(key,value);
localStorage.getItem(key);

方式二：通过 BrowserWindow 和 和 webContents 模块实现渲染进 程和渲染进程的通信

webContents 是一个事件发出者.它负责渲染并控制网页，也是BrowserWindow 对象的属性。

需要了解的几个知识点：

1 、获取当前窗口的 id：

const winId = BrowserWindow.getFocusedWindow().id;

2、监听当前窗口加载完成的事件

win.webContents.on('did-finish-load',(event) => { })

3、同一窗口之间广播数据

win.webContents.on('did-finish-load',(event) => {
    win.webContents.send('msg',winId,'我是 index.html 的数据');
})

4、通过 id 查找窗口

let win = BrowserWindow.fromId(winId);

    });
  }
  // 监听退出app
  listenerQuit() {
    ipcMain.on('quit', () => {
      app.quit();
    });
  }
}

import { ipcRenderer, remote } from 'electron';
import fs from 'fs';
import path from 'path';

const { dialog } = remote;

class EditPlus {
  constructor(id) {
    this.id = id;
    // 正在编辑的文件
    this.isEdit = false;
    // 正在编辑的文件路径
    this.filePath = '';
    // 新建文件名称
    this.newFileName = '无标题';
  }
  init() {
    // 获取dom
    this.getDom();
    // 创建右键菜单
    this.contextmenu();
    // 监听菜单的点击事件
    this.listenerAction();
  }
  getDom() {
    const self = this;
    this.editDom = document.querySelector(this.id);
    // 用户输入的时候，编辑状态
    this.editDom.addEventListener('input', () => {
      const fileName = self.filePath || self.newFileName;
      this.isEdit = true;
      // 设置标题
      self.editWindowName(`●${fileName}`);
    });
  }
  contextmenu() {
    document.addEventListener('contextmenu', () => {
      ipcRenderer.send('contextmenu');
    });
  }
  listenerAction() {
    const self = this;
    ipcRenderer.on('action', (event, data) => {
      switch (data) {
      case 'new':
          // 创建文件
        self.createFile();
        break;
      case 'open':
          // 打开文件
        self.openFile();
        break;
      case 'save':
        self.saveFile();
        break;
      case 'quit':
        self.quitFile();
        break;
      default:
        break;
      }
    });
  }
  createFile() {
    const self = this;
    if (this.isEdit) {
      this.handleQustionSave().then(() => {
        self.saveFile(() => {
          self.setEditAttr();
        });
      }).catch(() => {
        this.setEditAttr();
      });
    } else {
      this.setEditAttr();
    }
  }
  setEditAttr(path = '', value = '', isEdit = false) {
    const pathName = path || this.newFileName;
    // 改为非编辑模式
    this.isEdit = isEdit;
    // 清空内容
    this.editDom.value = value;
    // 清空文件路径
    this.filePath = path;
    // 设置标题
    this.editWindowName(pathName);
  }
  openFile() {
    const self = this;
    if (this.isEdit) {
      this.handleQustionSave().then(() => {
        self.saveFile(() => {
          self.setEditAttr();
        });
      }).catch(() => {
        this.handleOpenDialog();
      });
    } else {
      this.handleOpenDialog();
    }
  }
  // 触发打开文件弹窗
  handleOpenDialog() {
    const self = this;
    dialog.showOpenDialog(
      {
        title: '打开文件',
        properties: ['openFile'],
      },
      (res) => {
        // 点取消时，res为undefined,点确定为数组，单选路径取第一个
        if (res) {
          const path = res[0];
          fs.readFile(path, 'utf-8', (err, data) => {
            if (!err) {
              self.setEditAttr(path, data);
            } else {
              console.log(err);
            }
          });
        }
      },
    );
  }
  // 询问是否保存
  handleQustionSave() {
    const self = this;
    const filePath = this.filePath;
    return new Promise((resolve, reject) => {
      const text = filePath || this.newFileName;
      const index = dialog.showMessageBox({
        type: 'none',
        buttons: ['yes', 'no'],
        title: 'editPlus',
        detail: `是否将更改保存到${text}中？`,
      });
      if (index === 0) {
        resolve('ok');
      } else {
        reject('cancal');
      }
    });
  }
  saveFile(callback = function () {}) {
    const self = this;
    const content = self.editDom.value;
    const basename = self.filePath ? path.basename(self.filePath) : '*.txt';
    let callbackFn = null;
    if (callback) {
      callbackFn = callback;
    }
    // const fileName = self.filePath || ;
    // 存在文件路径，直接保存
    if (self.filePath) {
      self.writeFile(self.filePath, content, callbackFn);
      return;
    }
    // 不存在，弹出框选择地址保存
    dialog.showSaveDialog(
      {
        title: '保存文件',
        defaultPath: basename,
        filters: [{ name: 'All Files', extensions: ['*'] }],
      },
      (res) => {
        if (res) {
          self.writeFile(res, content, callbackFn);
        }
      },
    );
  }
  quitFile() {
    const self = this;
    if (this.isEdit) {
      this.handleQustionSave().then(() => {
        self.saveFile(() => {
          //  退出文件
          ipcRenderer.send('quit');
        });
      }).catch(() => {
        ipcRenderer.send('quit');
      });
    } else {
    //  退出文件
      ipcRenderer.send('quit');
    }
  }
  writeFile(res, content, callback) {
    const self = this;
    fs.writeFile(res, content, (err) => {
      if (err) {
        console.log(err);
      } else {
        // 保存成功，非编辑状态
        console.log('保存成功');
        self.isEdit = false;
        self.filePath = res;
        self.editWindowName(`${self.filePath}`);
        if (callback) {
          callback();
        }
      }
    });
  }
  editWindowName(title) {
    document.title = title;
  }
}
window.onload = () => {
  const editPlus = new EditPlus('#editContent');
  editPlus.init();
};
