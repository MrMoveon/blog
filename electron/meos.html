# 自定义最小化最大化关闭按钮

当我们采用无边框窗口的时候，需要自定义最小化、最大化、还原、关闭按钮

  • 通过remote.getCurrentWindow()获取当前窗口
  • 通过win.isMaximized()判断窗口是否最大化
  • 通过按钮点击设置窗口的显示

<template>
  <div class="tool-menu">
    <div class="tool-menu__min" @click="setMin" title="最小化">
      <a-icon type="minus" />
    </div>
    <div class="tool-menu__max" @click="setMaxToggle" :title="isMax?'还原':'最大化'">
      <a-icon type="block" v-if="isMax" />
      <a-icon type="border" v-else />
    </div>
    <div class="tool-menu__close" @click="setClose" title="关闭">
      <a-icon type="close" />
    </div>
  </div>
</template>

<script>
import {remote} from 'electron'
let win = {}
export default {
  name: 'ToolMenu',
  data () {
    return {
      isMax: false
    }
  },
  mounted () {
    this.init()
  },
  methods: {
    init () {
      // 1.获取当前窗口
      win = remote.getCurrentWindow()
      // 2.判断窗口是否最大化
      this.isMax = !!win.isMaximized()
    },
    setMin () {
      win.minimize()
    },
    setMaxToggle () {
      if (win.isMaximized()) {
        this.isMax = false
        win.restore()
      } else {
        this.isMax = true
        win.maximize()
      }
    },
    setClose () {
      win.close()
    }
  }
}
</script>

# 无边框窗口的拖动

当设置无边框窗口，｛frame: false,transparent: true｝会隐藏系统提供的窗口边框和标题栏，这时候我们需要在元素的css里添加 -webkit-app-region: drag;进行拖拽，

如果元素里有可点击元素，需要给可点击元素加上-webkit-app-region: no-drag;,不然无法点击。

# 获取盘符

通过 child_process 中的 exec方法，来执行获取盘符的命令，并进行处理

import {exec} from 'child_process'
getDisk () {
      var self = this
      exec('wmic logicaldisk get caption', (err, stdout, stderr) => {
        if (err || stderr) {
          console.log('root path open failed' + err + stderr)
          return
        }
        let con = stdout.toString().trim()
        let arr = con.split('\n')
        let disk = []
        arr.filter(item => {
          if (item.indexOf('Caption') === -1) {
            console.log(item.toString().slice(0, 2))
            disk.push(item.toString().slice(0, 2))
          }
        })
        self.disk = disk  // =>["C:","D:","E:","F:"]
      })
    }

# 打开文件的2种方式

# 1.exec 通过子进程的exec来执行命令或路径

缺点：命名中有特殊字符或空格，可能不会执行

import {exec} from 'child-process'
const path = 'C:\Users\longfei_du\Desktop\xx.png'
exec(path)

# 2.shell.openExternal(url)

以系统默认设置打开外部协议,文件，文件夹都可以打开（推荐）

import {remote} from 'electron'
const {shell} = remote
const path = 'C:\Users\longfei_du\Desktop\xx.png'
shell.openExternal(path)

# 常见问题

# 为什么没有 vue-devtools

在开发环境运行项目，如果缺少 vue-devtools，请检查你的终端，检查是否在安装过程中有任何可能的错误消息。electron-vue集成了electron-devtools-installer包，可以在运行时安装启动vue-devtools，在网络被墙可能会安装失败，提示Unable to install vue-devtools

解决方法：

自己下载vue-devtools插件，放到src/main/plugins/目录，讲插件后缀改为zip并解压到vue-devtools目录，修改index.dev.js，进行安装，修改如下：









 












 
 





/**
 * This file is used specifically and only for development. It installs
 * `electron-debug` & `vue-devtools`. There shouldn't be any need to
 *  modify this file, but it can be used to extend your development
 *  environment.
 */

/* eslint-disable */
import {  BrowserWindow } from 'electron'

// Install `electron-debug` with `devtron`
require('electron-debug')({ showDevTools: true })

// Install `vue-devtools`
require('electron').app.on('ready', () => {
  // let installExtension = require('electron-devtools-installer')
  // installExtension.default(installExtension.VUEJS_DEVTOOLS)
  //   .then(() => {})
  //   .catch(err => {
  //     console.log('Unable to install `vue-devtools`: \n', err)
  //   })
  var devtools = require('path').resolve(__dirname, 'plugins/vue-devtools')
  BrowserWindow.addDevToolsExtension(devtools);
})

// Require `main` process to boot app
require('./index')

# 调用vuex的方法不生效

是因为electron-vue脚手架引入了vuex-electron这个插件，其中使用了createSharedMutations用于多进程之间数据共享

在渲染器进程中调用您需要使用的操作dispatch或mapActions。不要使用commit，因为通过commit进程触发的操作不会在进程之间共享。

如果不需要多进程之间数据共享

解决办法：

1.去掉store/index.js中的createSharedMutations插件。

2.在主进程中再次引入store文件

# 调用exec 出现中文乱码

解决办法：使用Node去解码，先将encoding设置为buffer，然后使用iconv-lite模块解码

 
 

 
 


















const iconv = require('iconv-lite');
const { exec } = require('child_process');
// 查询win磁盘 信息
exec('WMIC LOGICALDISK WHERE MEDIATYPE='12' GET DESCRIPTION,DEVICEID,FILESYSTEM,SIZE,FREESPACE', { encoding: 'buffer' }, (error, stdout) => {
    let stdoutIconv = iconv.decode(stdout, 'cp936')
    let con = stdoutIconv.toString().trim()
    let disk = con.split('\n')
    var diskData = []
    var reg = /([^a-zA-Z0-9_\s]+)\s+([a-zA-Z]:)\s+[a-zA-Z]+\s+(\d+)\s+(\d+)/i
    disk.forEach(item => {
      var data = reg.exec(item)
      if (data) {
        diskData.push({
          zh: data[1],
          en: data[2],
          spaceAvailable: data[3],
          total: data[4]
        })
      }
    })
    console.log(diskData)// =>[{zh: "本地固定磁盘", en: "C:", spaceAvailable: "84684034048", total: "214853218304"},{zh: "本地固定磁盘", en: "D:", spaceAvailable: "405308760064", total: "429496725504"},{zh: "本地固定磁盘", en: "E:", spaceAvailable: "324562104320", total: "355327799296"}]
});

# electron出现require is not defined以及webview无法显示

近期升级到electron5.0之后，运行便会出现require is not defined的错误，并且webview也无法正常显示

由于新版将nodeIntegration和webviewTag默认值改成了false.所以将其设置为ture即可。

 mainWindow = new BrowserWindow({
    height: 1920,
    useContentSize: true,
    width: 600,
    webPreferences: {
     nodeIntegration: true,  // 解决require is not defined问题
     webviewTag: true  // 解决webview无法显示问题
    }
  })
