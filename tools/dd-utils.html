# 2. 方法的各种定义

# 2.1 函数

# call

export function call(fn,ctx,...args){
    // 如果指向的对象为undefined/null
    if(ctx===undefined || ctx ===null){
        // 那么就等于全局对象
        ctx = globalThis
    }
    // 给ctx添加一个临时方法 指向fn
    ctx._temp = fn
    // 执行临时方法，并传入参数
    var result = ctx._temp(...args)
    // 删除临时方法
    delete ctx._temp
    // 返回结果
    return result
}

# apply

export default function apply(fn, ctx, args) {
  // 如果ctx为undefined/null,this指向window
  if (ctx === undefined || ctx === null) {
    ctx = window;
  }
  // 给ctx添加一个临时方法，方法指向fn
  ctx._temp = fn;
  // 通过ctx来执行这个方法，也就会执行fn函数，此时fn中的this肯定指向ctx
  let result = ctx._temp(...args);
  // 删除ctx中的临时函数
  delete ctx._temp;
  // 返回fn的执行结果
  return result;
}

# bind

import call from "./call";

export default function bind(fn, ctx) {
  // bind返回的是一个新函数
  return function () {
    // 获取参数
    var args = arguments;
    // 通过call调用原函数，并指定this为ctx,然后传递参数
    return call(fn, ctx, ...args);
  };
}

# debounce

/**
 * 函数防抖
 * @description 创建一个防抖动函数，该函数会在用户停止触发后，延迟 `wait` 毫秒后执行 `fn`
 * @param {*} fn 函数
 * @param {*} wait 等待执行时间
 */
export default function debounce(fn, wait) {
  // 用来保存定时器的标识id
  var timerId = null;
  // 返回一个事件监听函数，也就是防抖函数
  return function (e) {
    var ctx = this;
    // 清除未执行的定时器任务
    if (timerId) {
      clearTimeout(timerId);
    }
    // 处理延迟wait时间后执行定时器任务
    timerId = setTimeout(function () {
      // 调用fn处理事件
      fn.call(ctx, e);
      // 执行后，清空标识id
      timerId = null;
    }, wait);
  };
}

# throttle

/**
 * 节流函数
 * @description 创建一个节流函数，每隔delay时间执行一次fn
 * @param {*} fn 函数
 * @param {*} delay 多少时间执行一次
 */
export default function throttle(fn, delay) {
  var prev = 0;
  return function (e) {
    var now = Date.now();
    if (now - prev > delay) {
      fn.call(this, e);
      prev = now;
    }
  };
}

# 2.2 数组

# map

/**
 * @description 创建一个新数组，其结果是该数组中的每个元素是调用一次提供的函数后的返回值
 * @param {*} array
 * @param {*} callback
 */
export default function map(array, callback) {
  let result = [];
  for (let i = 0; i < array.length; i++) {
    result.push(callback(array[i], i));
  }
  return result;
}

# filter

/**
 * @description filter()方法创建一个新数组, 其包含通过所提供函数实现的测试的所有元素。
 * @param {*} array
 * @param {*} callback [element,[,index]]
 */
export default function filter(array, callback) {
  let result = [];
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i)) {
      result.push(array[i]);
    }
  }
  return result;
}

# reduce

/**
 * @description reduce方法对数组中的每个元素执行一个由您提供的reducer函数(升序执行)，将其结果汇总为单个返回值。
 * @param {*} array
 * @param {*} callback 接受四个参数[accumulator,currentValue,currentIndex]
 * @param {*} initValue
 */
export default function reduce(array, callback, initValue) {
  if (!array.length) {
    throw new Error("Reduce of empty array with no initial value");
  }
  let result = initValue || array[0];
  for (let i = 0; i < array.length; i++) {
    result = callback(result, array[i], i);
  }
  return result;
}

# find

/**
 * @description  find() 方法返回数组中满足提供的测试函数的第一个元素的值。否则返回 undefined
 * @param {*} array
 * @param {*} callback
 */
export default function find(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i)) {
      return array[i];
    }
  }
  return undefined;
}

# findIndex

/**
 * @description findIndex()方法返回数组中满足提供的测试函数的第一个元素的索引。若没有找到对应元素则返回-1。
 * @param {*} array
 * @param {*} callback
 */
export default function findIndex(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i)) {
      return i;
    }
  }
  return -1;
}

# every

/**
 * @description every() 方法测试一个数组内的所有元素是否都能通过某个指定函数的测试。它返回一个布尔值。
 * @param {*} array
 * @param {*} callback
 */
export default function every(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (!callback(array[i], i)) {
      return false;
    }
  }
  return true;
}

# some

/**
 * @description some() 方法测试数组中是不是至少有1个元素通过了被提供的函数测试。它返回的是一个Boolean类型的值。
 * @param {*} array
 * @param {*} callback
 */
export default function some(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i)) {
      return true;
    }
  }
  return false;
}

# slice

/**
 * @description slice() 方法返回一个新的数组对象，这一对象是一个由 begin 和 end 决定的原数组的浅拷贝（包括 begin，不包括end）。原始数组不会被改变。
 * @param {*} array
 * @param {*} begin 提取起始处的索引
 * @param {*} end 提取终止处的索引
 */
export default function slice(array, begin, end) {
  var result = [];
  // 如果省略 begin，则 从索引 0 开始
  begin = begin || 0;
  // 如果该参数为负数，则表示从原数组中的倒数第几个元素开始提取
  if (begin < 0) {
    begin = array.length + begin;
  }
  // 如果 end 被省略，就等于数组的长度
  end = end || array.length;
  // 如果 begin 超出原数组的索引范围，则会返回空数组
  if (begin > array.length) {
    return [];
  }
  // 如果 end 大于数组的长度，那么end = 数组长度
  if (end > array.length) {
    end = array.length;
  }
  // 如果begin比end大返回空数组
  if (begin > end) {
    return [];
  }
  for (let i = begin; i < end; i++) {
    result.push(array[i]);
  }
  return result;
}

# concat

/**
 * @description concat()方法用于合并两个或多个数组。此方法不会更改现有数组，而是返回一个新数组
 * @param {*} array
 * @param  {...any} args 多个参数
 */
export default function concat(array, ...args) {
  let arr = [...array];
  for (let i = 0; i < args.length; i++) {
    const element = args[i];
    if (Array.isArray(element)) {
      arr.push(...element);
    } else {
      arr.push(element);
    }
  }
  return arr;
}

# chunck

/**
 * @description chunck() 数组分块函数，将一个数组分成几块
 * @param {*} array 数组
 * @param {*} size 数量
 */
export default function chunck(array, size) {
  var result = [];
  for (let i = 0; i < array.length; i++) {
    const element = array[i];
    // 获取每一块的key
    let key = Math.floor(i / size);
    // 如果不存在，
    if (!result[key]) {
      result[key] = [element];
    } else {
      result[key].push(element);
    }
  }
  return result;
}

# unique

/**
 * @description 通过双层循环去重，性能较差
 * @param {*} array
 */
export function unique1(array) {
  // 声明一个新数组
  let result = [];
  // 循环数组
  for (let i = 0; i < array.length; i++) {
    let current = array[i];
    // 通过indexOf检测当前项是否在result中，不在就插入到result
    if (result.indexOf(current) === -1) {
      result.push(array[i]);
    }
  }
  return result;
}

/**
 * @description 通过增加一个对象，进行去重，性能较好
 * @param {*} array
 */
export function unique2(array) {
  // 声明一个对象
  let obj = {};
  // 声明一个新数组
  let result = [];
  // 循环数组
  for (let i = 0; i < array.length; i++) {
    // 如果对象中没有这个key值，向新数组插入，并给对象添加这个key
    let key = array[i];
    if (!obj[key]) {
      result.push(key);
      obj[key] = true;
    }
  }
  return result;
}

/**
 * @description 通过Set的特性进行去重
 * @param {*} array
 */
export function unique3(array) {
  return [...new Set(array)];
}

export default {
  unique1,
  unique2,
  unique3,
};

# flatten

/**
 * @description flatten1() 数组通过递归进行扁平化
 * @param {*} array
 */
export function flatten1(array) {
  var arr = [...array];
  var result = [];
  for (let i = 0; i < arr.length; i++) {
    const element = arr[i];
    if (Array.isArray(element)) {
      result = result.concat(flatten(element));
    } else {
      result = result.concat(element);
    }
  }
  return result;
}

/**
 * @description flatten2() 数组通过递归进行扁平化
 * @param {*} array
 */
export function flatten2(array) {
  var arr = [...array];
  // 如果数组中存在数组，就对数组进行合并
  while (arr.some((item) => Array.isArray(item))) {
    // arr = [1,2,[3,4,[5,6]]
    // arr = [1,2,3,4,[5,6]]
    // arr = [1,2,3,4,5,6]
    arr = [].concat(...arr);
  }
  return arr;
}
export default {
  flatten1,
  flatten2,
};

# difference

/**
 * @description difference() 数组差集 创建一个具有唯一array值的数组，每个值不包含在其他给定的数组中
 * @param {Array} array 要检查的数组
 * @param {Array} [values] 排除的值
 */
export default function difference(array, values) {
  // 如果排除的值values 不存在，返回原数组
  if (!values || !values.length) return array;
  //  数组过滤，如果排除的值不包含当前值 返回当前值
  const result = array.filter((item) => !values.includes(item));
  return result;
}

# pull

/**
 * @description 移除数组array中所有和给定值相等的元素
 * @param {*} array 要修改的数组
 * @param  {...any} values 要移除的多个值
 */
export function pull(array, ...values) {
  if (!values.length) {
    return array;
  }
  const result = array.filter((item) => !values.includes(item));
  return result;
}

 */
export function pullAll(array, values = []) {
  return pull(array, ...values);

# drop

/**
 * @description drop() 创建一个切片数组，去除array前面的n个元素
 * @param {Array} array  要查询的数组
 * @param {Number} [size] 要去除的元素个数
 */
export function drop(array, size = 1) {
  // 通过过滤数组，判断数组的索引大于等于去除元素的个数
  const result = array.filter((item, index) => index >= size);
  return result;
}

drop([1, 2, 3]);
// => [2, 3]

drop([1, 2, 3], 2);
// => [3]


/**
 * @description dropRight() 创建一个切片数组，去除array尾部的n个元素
 * @param {Array} array  要查询的数组
 * @param {Number} [size] 要去除的元素个数
 */
export function dropRight(array, size = 1) {
  // 通过过滤数组，判断数组的索引 < 数组的长度-去除元素的个数
  const result = array.filter((item, index) => index < array.length - size);
  return result;
}
dropRight([1, 2, 3]);
// => [1, 2]

dropRight([1, 2, 3], 2);
// => [1]

dropRight([1, 2, 3], 5);
// => []

dropRight([1, 2, 3], 0);
// => [1, 2, 3]

/**
 * @description create() 创建一个继承 prototype 的对象
 * @param {Function} fn 要继承的对象
 * @param  {...any} props 待分配的属性
 */
export default function create(fn, ...props) {
  // 1.创建一个对象
  let obj = {};
  // 2.改变this执行
  const result = fn.call(obj, ...props);
  // 3.修改新对象的原型对象
  obj.__proto__ = fn.prototype;
  // 4.返回新对象
  return result instanceof Object ? result : obj;
}

# 4.5 测试覆盖率

先来了解一个概念code coverage,代码测试覆盖率，就是我们的测试代码，对功能性代码和业务逻辑代码作了百分多少的测试，这个百分比，就叫做代码测试覆盖率。

coverageDirectroy的配置是用来更改生成代码覆盖率文件的目录，

coverageDirectory : "coverage"

当这个选项被打开后，我们就可以使用下面的命令,jest就会自动给我们生成一个代码测试覆盖率的说明。

 npx jest --coverage

当然这个不仅会有一个简单的终端图表，还会生成一个coverage的文件夹，这里边有很多文件。

我们可以打开coverage>lcov-reporrt>index.html文件，这时候就可以看到一个网页形式的，非常漂亮的测试覆盖率报告。

那这个配置项可以作什么配置那？我们先来作一个修改，把coverageDirectory : "coverage" 改为coverageDirectory : "testReport",然后再使用npx jest --coverage,这时候你会发现，生成的报告文件夹，不再是coverage而是变成了testReport。

# 4.5 开启自动测试

每次修改测试用例，我们都手动输入npx test ，这显得很lower。可以通过配置package.json文件来设置。修改如下

"scripts": {
    "test": "jest",
    "coverage": "jest --coverage",
    "autotest": "jest --watchAll"
  },

修改保存后，我们在终端运行npx autotest,这时候测试一次后，它并没有结束，而是等待测试文件的变化，变化后就会自动进行测试。

# 4.6 jest中的匹配器

# toBe

toBe() 这个相当是等同于=== , 也就是我们常说的严格相等

# toEqual

toEqual()匹配器 可以把它理解成只要内容相等 就可以通过测试

# toStrictEqual

使用 . tostrictequal 来测试对象是否具有相同的类型和结构。

与.toEqual差异:

  1. 带有未定义属性的键被选中。例如，当使用. tostrictequal时，{a: undefined, b: 2}不匹配{b: 2}。
  2. 检查了数组稀疏性。例如:[，1]在使用. tostrictequal时不匹配[undefined, 1]。
  3. 对象类型被检查为相等。例如，一个有字段A和b的类实例将不等于一个有字段A和b的文字对象。
test("新数组的每一个值是原数组的值*10", () => {
  var a = [1, 2, 3, 4, 5];

  expect(
    map(a, function (item, index) {
      return item * 10;
    })
  ).toStrictEqual([10, 20, 30, 40, 50]);
});

# toBeNull

toBeNul()匹配器只匹配null值，不匹配undefined的值

# toBeUndefined

我们要匹配undefined时，就可以使用toBeUndifined()匹配器

test('toBeUndefined测试',()=>{
    const a = undefined   
    expect(a).toBeUndefined()
})

# toBeDefined

toBeDefined()匹配器的意思是只要定义过了，都可以匹配成功，例如下面的代码：

test('toBeDefined测试',()=>{
    const a = 'jspang'  
    expect(a).toBeDefined()
})

这里需要注意的是，我给一个null值也是可以通过测试的。

# toBeTruthy

这个是true和false匹配器，就相当于判断真假的。比如说写下面这样一段代码:

test('toBeTruthy 测试',()=>{
    const a = 0
    expect(a).toBeTruthy()
})

这样测试就过不去了，因为这里的0，如果判断真假时，就是false，所以无法通过。同样道理null也是无法通过的。 但是我们给个1或者abc字符串，就可以通过测试了。

# toBeFalsy

有真就有假，跟toBeTruthy()对应的就是toBeFalsy,这个匹配器只要是返回的false就可以通过测试。

test('toBeFalsy 测试',()=>{
    const a = 0
    expect(a).toBeFalsy()
})

# toBeGreaterThan

这个是用来作数字比较的，大于什么数值，只要大于传入的数值，就可以通过测试。我们来写一段代码来看一下。

test('toBeGreaterThan匹配器',()=>{
    const count = 10
    expect(count).toBeGreaterThan(9)
})

# toBeLessThan

toBeLessThan跟toBeGreaterThan相对应的，就是小于一个数字时，就可以通过测试。代码如下:

test('toBeLessThan匹配器',()=>{
    const count = 10
    expect(count).toBeLessThan(11)
})

10比11小，所以测试用例顺利通过。

# toBeGreaterThanOrEqual

当测试结果数据大于等于期待数字时，可以通过测试。比如下面的代码是没办法通过测试的。

test('toBeGreaterThan匹配器',()=>{
    const count = 10
    expect(count).toBeGreaterThan(10)
})

但是我们使用了toBeGreaterThanOrEqual()就可以通过测试。

test('toBeGreaterThanOrEqual匹配器',()=>{
    const count = 10
    expect(count).toBeGreaterThanOrEqual(10)
})

# toBeGreaterThanOrEqual

这个跟toBeGreaterThanOrEqual()相对应，所以就不做过多的介绍了。

# toBeCloseTo

这个是可以自动消除JavaScript浮点精度错误的匹配器，举个例子，比如我们让0.1和0.2相加，这时候js得到的值应该是0.30000000000004,所以如果用toEqual()匹配器，测试用例会通过不了测试的。

比如我们把测试用例写成这样，就不会通过:

test('toEqual匹配器',()=>{
    const one = 0.1
    const two = 0.2
    expect(one + two).toEqual(0.3)
})

这时候我们就需要使用toBeCloseTo()匹配器，可以顺利通过测试,代码如下：

test('toBeCloseTo匹配器',()=>{
    const one = 0.1
    const two = 0.2
    expect(one + two).toBeCloseTo(0.3)
})

这样就可以顺利通过测试了，这就是它的作用。

这节讲的匹配器都是跟数字有关的匹配器，可以动手做一下，但是我个人觉的没必要记忆，你只要知道有这些匹配器，在你的意识里有就可以了。当我们使用的多了，自然就记住了，并且会使用了。

# toMatch字符串包含匹配器

字符串包含匹配器，比如象牙山洗脚城有三个美女：谢大脚、刘英和小红，这时候我们要看看字符串中有没有谢大脚就可以使用toMatch()匹配器。

test('toMatch匹配器',()=>{
    const str="谢大脚、刘英、小红"
    expect(str).toMatch('谢大脚')
})

因为确实有“谢大脚”,所以就通过测试了，当然你也可以写正则表达式。

test('toMatch匹配器',()=>{
    const str="谢大脚、刘英、小红"
    expect(str).toMatch(/谢大脚/)
})

我们可以开启jest就可以看到测试结果也是正确的。

# toContain

刚才我们使用的只是一个字符串包换关系的匹配器，但是在工作中使用的多是数组，所以这里我们使用数组的匹配器toContain()。比如还是上面象牙山洗脚城的案例，我们就可以写成这样。

test('toContain匹配器',()=>{
    const arr=['谢大脚','刘英','小红']
    expect(arr).toContain('谢大脚')
})

当然他也可以完美的兼容set的测试，比如把下面代码改为下面的方式。

test('toContain匹配器',()=>{
    const arr=['谢大脚','刘英','小红']
    const data = new Set(arr)
    expect(data).toContain('谢大脚')
})

# toThrow

专门对异常进行处理的匹配器，可以检测一个方法会不会抛出异常。比如我们下面一段代码。

const throwNewErrorFunc = ()=>{
    throw new Error('this is a new error')
}

test('toThrow匹配器',()=>{
    expect(throwNewErrorFunc).toThrow()
})

我们也可以对这个匹配器中加一些字符串，意思就是抛出的异常必须和字符串相对应。

const throwNewErrorFunc = ()=>{
    throw new Error('this is a new error')
}

test('toThrow匹配器',()=>{
    expect(throwNewErrorFunc).toThrow('this is a new error')
})

如果字符串不匹配，也没办法通过异常测试。

# not

not匹配器是Jest中比较特殊的匹配器，意思就是相反或者说取反.比如上面的例子，我们不希望方法抛出异常，就可以使用not匹配器。

const throwNewErrorFunc = ()=>{
    throw new Error('this is a new error')
}

test('toThrow匹配器',()=>{
    expect(throwNewErrorFunc).not.toThrow()
})

现在这个测试用例就不能通过测试了，我们需要删除或注释掉抛出的异常，才可以通过测试。这就是not匹配器的用法。
