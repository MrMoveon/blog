var temp = "";
var arr = [];

  var current = str.charAt(i);
  if (Number(current) < 10) {
    temp += current;
  } else {
    if (temp) {
      arr.push(temp.trim());
      temp = "";
    }
  }
}
if (temp) {
  arr.push(temp.trim());
  temp = "";
}

# 转义字符

  • . —— 任意字符
  • \d 数字 [0-9]
  • \w 英文、数字、下划线 [a-z0-9_]
  • \s 空白字符
  • \D 除了数字 [^0-9]
  • \W 除了英文、数字、下划线 [^a-z0-9_]
  • \S 除了空白字符

# 量词

什么是量词？ 就是出现的次数。

  • {n} 出现 n 次
  • {n,m} 出现最少 n 次，最多 m 次
  • {n,} 至少 n 次
  • *任意次{0,}
  • ？出现 0 次或 1 次{0,1}
  • +一次或任意次{1,}

# 量词实例：匹配固定电话

常用固定电话格式：010-3536424 、0101-3536424-12345、3536424

分析：

  • 区号以 0 开头，可以有 3-4 位的区号，并且区号可有可无

  • 电话号码，不能以 0 开头，可以有 7-8 位的号码

  • 分机：1-5 位，可有可无

var t1 = "010-3536424";
var t2 = "0101-3536424-12345";
var t3 = "3536424";
var reg = /(0\d{2-3}-)?[1-9]\d{6,7}(-\d{1,5})?/;
console.log(reg.test(t1)); // => true
console.log(reg.test(t2)); // => true
console.log(reg.test(t3)); // => true

# 实例：验证邮箱

分析：w345848848@qq.com 邮箱是由 一串字母、数字、下划线开头（w345848848） 加上（@） 符号 加上一串字母、数字（qq） 加上 （.）加上一串字母结尾(com)组成

[^] 在中括号中取反

^ 以什么开头

$ 以什么结尾

var email = "w345848848@qq.com";
var reg = /^\w+@[a-z0-9]+\.[a-z]+$/i;
reg.test(email);

# 实例：格式化数据

思路，先匹配出以下的每个字符串，再通过格式插入到数组中

var disk = [
  "本地固定磁盘  C:        NTFS        83702444032   214853218304",
  "本地固定磁盘  D:        NTFS        405313318912  429496725504",
  "本地固定磁盘  E:        NTFS        325011607552  355327799296"
];
var diskData = [];
var reg = /([^a-zA-Z0-9_\s]+)\s+([a-zA-Z]:)\s+[a-zA-Z]+\s+(\d+)\s+(\d+)/i;
disk.forEach(item => {
  var data = reg.exec(item);
  diskData.push({
    zh: data[1],
    en: data[2],
    spaceAvailable: data[3],
    total: data[4]
  });
});
console.log(diskData);
// => [{zh: "本地固定磁盘", en: "C:", spaceAvailable: "83702444032", total: "214853218304"},{zh: "本地固定磁盘", en: "D:", spaceAvailable: "405313318912", total: "429496725504"},{zh: "本地固定磁盘", en: "E:", spaceAvailable: "325011607552", total: "355327799296"}]

# 实例：提取内容

提取["POWER"] 中的POWER, 注意[]""需要进行\转译

var str = '["POWER"]';
var regexp = /^\[\"(\w+)\"\]$/;
var result = regexp.exec(str) // ["["POWER"]", "POWER", index: 0, input: "["POWER"]"]
console.log(result[1]) // POWER
