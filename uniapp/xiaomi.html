      # 动画库使用和图标库创建

      animate.css 预置了几十种 css动画效果，只需要在标签上添加class.

      <text class="title bounceIn">{{title}}</text>

      uni-app支持使用字体图标，使用方式与普通 web 项目相同，需要注意以下几点：

      • 支持 base64 格式字体图标。
      • 支持网络路径字体图标。
      • 网络路径必须加协议头 https。
      • 从 http://www.iconfont.cn 上拷贝的代码，默认是没加协议头的。
      • uni-app 本地路径图标字体支持情况：
        1. 字体文件小于 40kb，uni-app 会自动将其转化为 base64 格式；
        2. 字体文件大于等于 40kb， 需开发者自己转换，否则使用将不生效；
        3. 字体文件的引用路径仅支持以 ~@ 开头的绝对路径（不支持相对路径）。
        4. 图标显示兼容性最好使用unicode的方式，用font class部分设备不能显示

      iconfont 中进行选择图标创建，将图标下载至本地，打开iconfont.css,删除字体文件的路径，只保留base64格式字体或路径使用远程地址并加上协议头https，代码如下：

      @font-face {
	font-family: "iconfont";
	src: url('https://at.alicdn.com/t/font_1273755_07yj85s3ope5.ttf') format('truetype')
    /*src: url('base64') format('truetype')*/
}

      # 滚动导航栏开发

      通过scroll-view组件，实现点击滚动居中显示效果

      <template>
	<view class="home">
		<scroll-view 
			id="tab-bar" 
			class="uni-swiper-tab" 
			scroll-x 
			scroll-with-animation 
			:scroll-left="scrollLeft">
			<view 
				v-for="(tab,index) in tabBars" 
				:key="tab.id" 
				class="swiper-tab-list" 
				:class="tabIndex==index ? 'active' : ''"
				:id="tab.id" 
				:data-current="index" 
				@click="tapTab">
			{{tab.name}}</view>
		</scroll-view>
	</view>
</template>

<script>
	export default {
		data() {
			return {
				winWidth:0,
				centerPoint:0,
				tabBarWidth:0,
				scrollLeft: 0,
				tabIndex: 0,
				tabBars: [{
					name: '关注',
					id: 'guanzhu'
				}, {
					name: '推荐',
					id: 'tuijian'
				}, {
					name: '体育',
					id: 'tiyu'
				}, {
					name: '热点',
					id: 'redian'
				}, {
					name: '财经',
					id: 'caijing'
				}, {
					name: '娱乐',
					id: 'yule'
				}, {
					name: '军事',
					id: 'junshi'
				}, {
					name: '历史',
					id: 'lishi'
				}, {
					name: '本地',
					id: 'bendi'
				}]
			}
		},
		async onReady() {
			await this.initNav()
		},
		
		methods: {
			async initNav(){
				// 获取屏幕宽度
				this.winWidth = uni.getSystemInfoSync().windowWidth;
				// 获取中心点x
				this.centerPoint = this.winWidth/2;
				
				let width = 0;
				let index = this.tabBars.length;
				for (let i = 0; i < index; i++) {
					let result = await this.getElSize(this.tabBars[i].id);
					// 获取每个元素的宽度
					this.tabBars[i].width = result.width
					width += result.width;
				}
				// tabBar宽度
				this.tabBarWidth = width
			},
			getElSize(id) { //得到元素的size
				return new Promise((res, rej) => {
					uni.createSelectorQuery().select("#" + id).fields({
						size: true,
						scrollOffset: true
					}, (data) => {
						res(data);
					}).exec();
				})
			},
			async tapTab(e) { //点击tab-bar
				let tabIndex = Number(e.target.dataset.current);
				if (this.tabIndex === tabIndex) {
					return false;
				} else {
					let width = 0;
					for (let i = 0; i <= tabIndex; i++) {
						let result = this.tabBars[i];
						// 获取第一个元素到点击元素的宽度
						width += result.width;
						if(i===tabIndex){
							if(width>=this.centerPoint){
								// 居中显示 
								this.scrollLeft = width-this.centerPoint-result.width/2
							}else{
								// left
								this.scrollLeft = 0;
							}
							this.tabIndex = tabIndex;
							return
						}
					}
				}
			}
		
		}
	}
</script>

      <template>
	<swiper class="swiper-image" 
		indicator-dots 
		autoplay 
		circular 
		indicator-color="rgba(255,255,255,0.5)"

	</swiper>
</template>

<script>
	export default {
		name:"swiper-image",
		props:{
			swiperData:{
				type:Array,
				default:function(){
					return []
				}
			}
		},
		methods:{
			handle(item){
				console.log(item.url);
			}
		}
	}
</script>
