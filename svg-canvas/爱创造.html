        # 拖拽类的编写

        class Drag {
	constructor(props) {
		Object.assign(this, {
			el: null, // dom对象
			pageX:0,
			pageY:0,
			orgiX: 0,
			orgiY: 0,
			moveX: 0,
			moveY: 0,
			startFn: function() {},
			moveFn: function() {},
			endFn: function() {}
		}, props);
		
		this.bindEvent()

	}
	bindEvent(){
 		// bind返回的是一个新的不同的函数，removeEventListener此时无法实现事件移除，
		// 正确方式如下:
		this.handleDown = this.handleDown.bind(this)
		this.handleMove = this.handleMove.bind(this)
		this.handleUp = this.handleUp.bind(this)
		
		this.el.addEventListener('mousedown', this.handleDown)
	}
	handleDown(ev){
		ev.preventDefault()
		this.pageX = ev.pageX;
		this.pageY = ev.pageY;
		this.startFn && this.startFn();
		document.addEventListener('mousemove', this.handleMove)
		document.addEventListener('mouseup', this .handleUp)
	}
	handleMove(ev){
		this.moveX = ev.pageX - this.pageX + this.orgiX;
		this.moveY = ev.pageY - this.pageY + this.orgiY;
		this.moveFn && this.moveFn(this.moveX, this.moveY)
	}
	handleUp(){
		var that = this
		this.orgiX = this.moveX
		this.orgiY = this.moveY
		this.endFn && this.endFn()
		
		document.removeEventListener('mousemove', that.handleMove)
		document.removeEventListener('mouseup', that.handleUp)
		console.log('结束');
	}
}

        # 采用发布-订阅自定义事件重写拖拽类

        
/**
 * 自定义事件
 */
class CustomEvent {
	constructor() {
		this.handle = {}
	}
	on(eventName, fn) {
		this.handle[eventName] = this.handle[eventName] || []
		this.handle[eventName].push(fn)
	}
	off(eventName, fn) {
		var events = this.handle[eventName] || []
		if (events.length) {
			for (var key in events) {
				if (events[key] === fn) {
					events.splice(key, 1)
				}
			}
		}
	}
	emit(eventName, ...params) {
		this.handle[eventName] = this.handle[eventName] || []
		this.handle[eventName].forEach(function(fn) {
			fn.call(this, ...params)
		})
	}
}

class Drag extends CustomEvent {
	constructor(props) {
		super()
		Object.assign(this, {
			el: null, // dom对象
			pageX: 0,
			pageY: 0,
			orgiX: 0,
			orgiY: 0,
			moveX: 0,
			moveY: 0,
		}, props);
		this.bindEvent()
	}
	bindEvent() {
		// bind返回的是一个新的不同的函数，removeEventListener此时无法实现事件移除，
		// 正确方式如下:
		this.handleDown = this.handleDown.bind(this)
		this.handleMove = this.handleMove.bind(this)
		this.handleUp = this.handleUp.bind(this)

		this.el.addEventListener('mousedown', this.handleDown)
	}
	handleDown(ev) {
		ev.preventDefault()
		this.pageX = ev.pageX;
		this.pageY = ev.pageY;
		// 触发拖拽开始事件
		this.emit('dragStart')
		document.addEventListener('mousemove', this.handleMove)
		document.addEventListener('mouseup', this.handleUp)
	}
	handleMove(ev) {
		this.moveX = ev.pageX - this.pageX + this.orgiX;
		this.moveY = ev.pageY - this.pageY + this.orgiY;
		// 触发拖拽移动事件
		this.emit('dragMove', this.moveX, this.moveY)
	}
	handleUp() {
		var that = this
		this.orgiX = this.moveX
		this.orgiY = this.moveY
		// 触发拖拽结束事件
		this.emit('dragEnd')
		document.removeEventListener('mousemove', that.handleMove)
		document.removeEventListener('mouseup', that.handleUp)
		console.log('结束');
	}
	
}

        # 按住空格拖拽

        
class Editor {
	constructor(props) {
	    Object.assign(this,{
			...
			isCanDrag:false, // 按住空格，是否可以拖动
		},props)
 		this.render()
		this.renderRulerBar()
		return this
	}
    render(){
        ...
        this.bindScaleDragEvent()
    }
	// 绑定编辑区缩放拖拽事件
	bindScaleDragEvent(){
		var that = this
		// 编辑区
		var editor = document.querySelector('.'+EDITOR_CLASS)
		// 缩放拖拽区域
		var scaleDrag =  document.querySelector('.'+SCALE_DRAG_CLASS)
		// 内容区
		var drawingBoard = document.querySelector('.'+DRAWINGBOARD_CLASS)
		// 监听键盘事件
		window.addEventListener('keydown',function(ev){
			// 如果是按住空格键，改变scaleDrag的鼠标样式，并可以触发事件侦听，
			if(ev.keyCode===32){
				if(!that.isCanDrag){
					scaleDrag.css({
						'pointer-events':'auto',
						'cursor':'-webkit-grab'
					})
				}
				that.isCanDrag = true
			} 
		})
		// 抬起按键，设置不能拖动，禁止触发事件侦听
		window.addEventListener('keyup',function(){
			that.isCanDrag = false
			scaleDrag.css({
				'pointer-events':'none',
				'cursor':'none '
			})
		})
		
		var editorDrag = new Drag({el:scaleDrag })  
		
		editorDrag.on('dragStart',function(){
			if(that.isCanDrag){
				scaleDrag.css({
					'pointer-events':'auto',
					'cursor':'-webkit-grabbing'
				})
			}
		})
		editorDrag.on('dragMove',function(moveX,moveY){
			// 设置拖动的值
			drawingBoard.css({
				marginTop:moveY+'px',
				marginLeft:moveX+'px',
			})
			// 重置标尺
			that.rulerBar.resetRuler()
		})
		editorDrag.on('dragEnd',function(){
			that.isCanDrag = false;
			scaleDrag.css({
				'pointer-events':'none',
				'cursor':'none '
			})
		})
	}
}

        # 常见问题

        # 事件添加与移除：bind()后无法移除事件问题

        bind返回的是一个新的不同的函数，此时也无法实现事件移除

        // 添加事件
document.addEventListener('mousemove', this.handleMove.bind(this))
document.addEventListener('mouseup', this .handleUp.bind(this))
// 移除事件，不生效
document.removeEventListener('mousemove', this.handleMove.bind(this))
document.removeEventListener('mouseup', this .handleUp.bind(this))

        正确方式如下 ：

        // 将事件方法提出来，重新赋值bind的方法
this.handleMove = this.handleMove.bind(this)
this.handleUp = this.handleUp.bind(this)
// 添加事件
document.addEventListener('mousemove', this.handleMove)
document.addEventListener('mouseup', this .handleUp)
// 移除事件，生效
document.removeEventListener('mousemove', that.handleMove)
document.removeEventListener('mouseup', that.handleUp)
